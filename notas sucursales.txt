		CODOFI,

		NIT, 
		NUMDOC, 
		NROCMP, 
		(ANOELB*10000+MESELB*100+DIAELB) as FECHA, 
		VALOR

	and ANOELB = 2025 
	and MESELB between 01 and 06 
	and DIAELB between 01 and 31
